CREATE TABLE autor (
autor_id int primary key,
nome varchar (255),
nacionalidade varchar (255)

);

CREATE TABLE Livros (
livro_id int primary key,
autor_id2 int,
titulo varchar (255),
ano_publicacao date,
foreign key (autor_id2) references Autor(autor_id)

);

CREATE TABLE Categorias (
categoria_id int primary key,
nome varchar (255)

);

CREATE TABLE Livros_Categorias (
livroca_id int primary key,
categoria_id2 int,
livro_id2 int,
foreign key (livro_id2) references Livros(livro_id),
foreign key (categoria_id2) references Categorias(categoria_id)

);

CREATE TABLE Usuários (
usuario_id int primary key,
nome varchar (255),
email varchar (255)

);

CREATE TABLE Emprestimos (
emprestimo_id int primary key,
livro_id2 int,
usuario_id2 int,
foreign key (livro_id2) references livros(livro_id),
foreign key (usuario_id2) references Usuários(usuario_id),
data_emprestimo date,
data_devolucao date

);
