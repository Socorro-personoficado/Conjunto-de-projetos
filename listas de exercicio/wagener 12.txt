using System.Windows.Forms;

namespace WinFormsApp1
{
    public partial class Form1 : Form
    {
        public Form1()
        {
            InitializeComponent();
        }
        // Evento
        private void button1_Click(object sender, EventArgs e)
        {

            string res = textBox1.Text;
            int compri = res.Length;
            if (compri % 2 == 1)
            {
                string parte1 = res.Substring(0, (compri / 2) + 1);
                string parte2 = res.Substring(compri / 2, compri - (compri / 2));
                textBox1.Text = parte1;
                textBox2.Text = parte2;
            }
            else
            {
                string parte1 = res.Substring(0, compri / 2);
                string parte2 = res.Substring(compri / 2, compri / 2);
                textBox1.Text = parte1;
                textBox2.Text = parte2;

                int x = 0;
                int y = 1;
                while (x < compri/2)
                {
                    if (parte2.Substring(parte2.Length - y, 1) == parte1.Substring(x, 1))
                    {
                        x++;
                        y++;
                        
                    }
                    else
                    {
                        label1.Text = "nao palindromo";
                        break;
                        
                    }
                    label1.Text = "palindromo";
                }
            } 

        }
    }
