        private void button1_Click(object sender, EventArgs e)
        {
            DateTime entrada = DateTime.Parse(textBox2.Text);
            DateTime saida = DateTime.Parse(textBox1.Text);
            DateTime horasT = saida.AddHours(-entrada.Hour);
            DateTime minuteT = saida.AddMinutes(-entrada.Minute);
            label6.Text = Convert.ToString(horasT.Hour)+" horas e "+ Convert.ToString(minuteT.Minute) + " minutos";
            if (horasT.Hour > 8)
            {
                DateTime horasEx = horasT.AddHours(-8);
                label5.Text = Convert.ToString(horasEx.Hour)+ " horas e " + Convert.ToString(minuteT.Minute) + " minutos";
            }

        }
    }
